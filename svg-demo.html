Usage Example

// Import the functions
import { glyphToSVG, templateToSVG } from './glyph-renderer';

// Generate a glyph SVG
const svgString = glyphToSVG(0b111111, {
  strokeWidth: 24,
  strokeColor: '#000000'
});

// Inject into DOM
document.getElementById('container').innerHTML = svgString;

// Or use as img src with data URI
const dataUri = 'data:image/svg+xml,' + encodeURIComponent(svgString);
img.src = dataUri;
